"""
chatgpt_web.py — Provider that drives ChatGPT via the DSL scripting engine.

The mediator server handles all browser interaction. This provider just:
1. Ensures the script session is started when activated.
2. Forwards requests to the local OpenAI-compat endpoint the mediator exposes.
"""

import json
import requests
from typing import Generator

PROVIDER_NAME = "chatgpt_web"
SCRIPT_ID     = "chatgpt"

MODELS = [
    {"id": "chatgpt",    "label": "ChatGPT (Web)",    "ctx": 32000, "script_id": SCRIPT_ID},
    {"id": "chatgpt-4o", "label": "ChatGPT 4o (Web)", "ctx": 32000, "script_id": SCRIPT_ID},
]

# Flask app base — same process, so we just POST internally
_INTERNAL_BASE = "http://localhost:5000"


def _ensure_session():
    """Start the mediator script session if not already running."""
    try:
        resp = requests.get(f"{_INTERNAL_BASE}/mediator/sessions", timeout=5)
        sessions = resp.json().get("sessions", [])
        active_ids = {s["id"] for s in sessions if s.get("loaded")}
        if SCRIPT_ID not in active_ids:
            requests.post(f"{_INTERNAL_BASE}/mediator/start/{SCRIPT_ID}", timeout=5)
    except Exception:
        pass  # Best-effort; mediator will start on first chat request anyway


def stream_chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> Generator[dict, None, None]:
    _ensure_session()

    payload = {
        "model":    SCRIPT_ID,
        "messages": messages,
    }
    if tools:
        payload["tools"] = tools

    try:
        resp = requests.post(
            f"{_INTERNAL_BASE}/v1/chat/completions/{SCRIPT_ID}",
            json=payload,
            timeout=180,
        )
        resp.raise_for_status()
        data = resp.json()
    except Exception as e:
        yield {"type": "error", "text": str(e)}
        return

    choice = (data.get("choices") or [{}])[0]
    msg    = choice.get("message", {})

    text = msg.get("content") or ""
    if text:
        yield {"type": "text", "text": text}

    for tc in msg.get("tool_calls") or []:
        fn   = tc.get("function", {})
        args = fn.get("arguments", "{}")
        try:
            args_parsed = json.loads(args)
        except Exception:
            args_parsed = {}
        yield {
            "type":      "tool_delta",
            "index":     0,
            "id":        tc.get("id", ""),
            "name":      fn.get("name", ""),
            "arguments": args,
        }

    yield {"type": "done"}
