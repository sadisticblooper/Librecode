"""
demo.py — Provider for the bundled demo site.

Uses the DSL executor to drive the demo WebView, with the same text-based
tool-calling scheme as the qwen provider (<tool_calls>...</tool_calls>).

Message handling
----------------
Only the last user message is typed into the site.  The system prompt and
conversation history are prepended as a compact context block so the demo
site sees them as part of the typed input — matching how a real provider
would receive the full context.

Tool calling
------------
If tools are passed, the system prompt is extended with tool instructions
(same format as qwen).  The reply is scanned for a <tool_calls> block and
parsed into tool_delta events before the final done event.
"""

import json

PROVIDER_NAME = "demo"
SCRIPT_ID     = "demo"

MODELS = [
    {"id": "demo", "label": "Demo Site (test)", "ctx": 4000, "script_id": SCRIPT_ID},
]

_TOOL_CALL_OPEN  = "<tool_calls>"
_TOOL_CALL_CLOSE = "</tool_calls>"


# ── Tool prompt injection (mirrors qwen provider) ─────────────────────────────

def _build_tool_system(tools: list) -> str:
    tool_specs = []
    for t in tools:
        fn       = t.get("function", t)
        name     = fn.get("name", "")
        desc     = fn.get("description", "")
        params   = fn.get("parameters", {})
        props    = params.get("properties", {})
        required = params.get("required", [])
        args_doc = []
        for k, v in props.items():
            req   = " (required)" if k in required else ""
            typ   = v.get("type", "any")
            pdesc = v.get("description", "")
            args_doc.append(f"  - {k} ({typ}){req}: {pdesc}")
        tool_specs.append(
            f"### {name}\n{desc}\nArguments:\n"
            + ("\n".join(args_doc) if args_doc else "  (none)")
        )

    return (
        "\n\n---\n"
        "## Tool Calling\n\n"
        "You have access to tools. If you need to call one or more tools, "
        "output your full response text first, then append a tool call block "
        "as the VERY LAST thing in your message — no text after it.\n\n"
        "Format (strict — no markdown fences, no extra whitespace before the tag):\n"
        "<tool_calls>\n"
        '[{"name": "tool_name", "args": {"arg1": "val1"}}, ...]\n'
        "</tool_calls>\n\n"
        "Rules:\n"
        "- Only include the block if you are actually calling a tool.\n"
        "- All tool calls for a single turn go in one block as a JSON array.\n"
        "- args values must match the expected types exactly.\n"
        "- After you receive tool results they will appear as user messages; "
        "continue your response naturally.\n\n"
        "Available tools:\n\n"
        + "\n\n".join(tool_specs)
    )


def _build_input(messages: list, tools: list | None) -> str:
    """
    Build the string that gets typed into the demo site input box.

    Format:
      [System]: <system prompt + optional tool instructions>
      [User]: <msg>
      [Assistant]: <msg>
      ...
      [User]: <last user message>   ← this is what the site "responds" to
    """
    # Merge tool instructions into system prompt if needed
    sys_extra = _build_tool_system(tools) if tools else ""

    lines = []
    for m in messages:
        role    = m.get("role", "")
        content = m.get("content") or ""
        if isinstance(content, list):
            # multi-part content — join text parts
            content = " ".join(
                p.get("text", "") for p in content if p.get("type") == "text"
            )

        if role == "system":
            lines.append(f"[System]: {content}{sys_extra}")
        elif role == "user":
            lines.append(f"[User]: {content}")
        elif role == "assistant":
            lines.append(f"[Assistant]: {content}")
        elif role == "tool":
            # tool result — surface it as a bracketed block
            lines.append(f"[Tool result]: {content}")

    return "\n".join(lines).strip()


# ── Tool call parsing (mirrors qwen provider) ─────────────────────────────────

def _parse_tool_calls(reply: str):
    """
    Returns (clean_text, tool_calls_list).
    clean_text has the <tool_calls> block stripped.
    tool_calls_list is [] if no block found or parse fails.
    """
    if _TOOL_CALL_OPEN not in reply:
        return reply, []

    before, _, after = reply.partition(_TOOL_CALL_OPEN)
    raw_block = after.split(_TOOL_CALL_CLOSE)[0].strip()
    clean     = before.strip()

    try:
        parsed = json.loads(raw_block)
        if not isinstance(parsed, list):
            raise ValueError("expected JSON array")
        return clean, parsed
    except Exception:
        # Malformed block — return full reply, no tool calls
        return reply, []


# ── Provider entry point ──────────────────────────────────────────────────────

def stream_chat(model_id, messages, tools=None, max_tokens=None):
    from python.mediator.mediator_server import _get_executor, _load_manifest
    from python.mediator.script_session import session_manager
    from python.mediator.dsl_executor import DslError

    manifest = _load_manifest(SCRIPT_ID)
    if not manifest:
        yield {"type": "error", "text": "No manifest found for demo script"}
        return

    executor = _get_executor(SCRIPT_ID)
    if not executor:
        yield {"type": "error", "text": "Failed to load demo executor"}
        return

    user_input = _build_input(messages, tools)

    try:
        if not session_manager.is_loaded(SCRIPT_ID):
            session_manager.get_or_create(
                script_id=SCRIPT_ID,
                url=manifest.get("url", ""),
                port=manifest.get("port", 0),
            )
            executor.load()
            session_manager.mark_loaded(SCRIPT_ID)

        executor.send(user_input)
        reply = executor.read()
        session_manager.reset_errors(SCRIPT_ID)

    except DslError as e:
        yield {"type": "error", "text": f"DSL error: {e}"}
        return
    except Exception as e:
        yield {"type": "error", "text": f"Executor error: {e}"}
        return

    if not reply:
        yield {"type": "done"}
        return

    clean, tool_calls = _parse_tool_calls(reply)

    if clean:
        yield {"type": "text", "text": clean}

    for i, tc in enumerate(tool_calls):
        name = tc.get("name", "")
        args = tc.get("args", tc.get("arguments", {}))
        yield {
            "type":      "tool_delta",
            "index":     i,
            "id":        "",
            "name":      name,
            "arguments": json.dumps(args) if not isinstance(args, str) else args,
        }

    yield {"type": "done"}
