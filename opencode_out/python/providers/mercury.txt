import requests
from typing import Generator

API_KEY       = "sk_ed6c5f9b76c20217c5b95a6c954d916f"

MODELS = [
    {"id": "mercury-2",      "label": "Mercury 2",      "ctx": 32000},
]

# Only mercury-2 supports tools
_TOOLS_SUPPORTED = {"mercury-2"}


def stream_chat(
    model_id:   str,
    messages:   list[dict],
    tools:      list | None = None,
    max_tokens: int  | None = None,
) -> Generator[dict, None, None]:

    url     = f"{API_BASE}/chat/completions"
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type":  "application/json",
    }
    payload = {
        "model":      model_id,
        "messages":   messages,
        "stream":     True,
        "max_tokens": min(max_tokens or 4096, 32000),
    }
    if tools and model_id in _TOOLS_SUPPORTED:
        payload["tools"]       = tools
        payload["tool_choice"] = "auto"

    api_resp = requests.post(url, json=payload, headers=headers, stream=True, timeout=(15, 120))
    if not api_resp.ok:
        yield {"type": "error", "text": f"HTTP {api_resp.status_code}: {api_resp.text}"}
        return

    tool_calls_buf: dict[int, dict] = {}

    try:
        for raw in api_resp.iter_lines():
            if not raw:
                continue
            line = raw.decode("utf-8", errors="replace")
            if not line.startswith("data: "):
                continue
            data_str = line[6:].strip()
            if data_str == "[DONE]":
                break
            try:
                chunk = json.loads(data_str)
            except json.JSONDecodeError:
                continue

            choice = (chunk.get("choices") or [{}])[0]
            delta  = choice.get("delta", {})

            content = delta.get("content") or ""
            if content:
                yield {"type": "text", "text": content}

            for tc in delta.get("tool_calls") or []:
                idx = tc.get("index", 0)
                fn  = tc.get("function", {})
                if idx not in tool_calls_buf:
                    tool_calls_buf[idx] = {"id": tc.get("id", ""), "name": fn.get("name", ""), "arguments": ""}
                else:
                    if tc.get("id"):   tool_calls_buf[idx]["id"]   = tc["id"]
                    if fn.get("name"): tool_calls_buf[idx]["name"] += fn["name"]
                tool_calls_buf[idx]["arguments"] += fn.get("arguments", "")

    except Exception as e:
        yield {"type": "error", "text": f"Stream error: {e}"}
        return

    for idx in sorted(tool_calls_buf):
        tc = tool_calls_buf[idx]
        yield {"type": "tool_delta", "index": idx, "id": tc["id"], "name": tc["name"], "arguments": tc["arguments"]}

    yield {"type": "done"}
