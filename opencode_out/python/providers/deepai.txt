# ── Tool prompts ───────────────────────────────────────────────────────

def _build_tool_system_small(tools: list) -> str:
    """
    Extra-verbose prompt for small/weaker models.
    Shows the full expected input/output sequence step by step.
    """
    tool_specs = []
    for t in tools:
        fn       = t.get("function", t)
        name     = fn.get("name", "")
        desc     = fn.get("description", "")
        params   = fn.get("parameters", {})
        props    = params.get("properties", {})
        required = params.get("required", [])
        arg_lines = []
        for k, v in props.items():
            req   = " (REQUIRED)" if k in required else " (optional)"
            typ   = v.get("type", "string")
            pdesc = v.get("description", "")
            arg_lines.append(f'    "{k}": <{typ}>{req}  // {pdesc}')
        schema = (
            "{\n"
            + ",\n".join(arg_lines)
            + "\n  }"
        ) if arg_lines else "{}"
        tool_specs.append(f"Tool name: {name}\nWhat it does: {desc}\nArgument schema:\n  {schema}")

    first_tool = tools[0].get("function", tools[0]).get("name", "some_tool") if tools else "some_tool"

    return (
        "\n\n---\n"
        "## HOW TO USE TOOLS\n\n"
        "You have tools available. A tool lets you read files, run commands, write code, etc.\n\n"
        "STEP 1 — When you need to use a tool, output ONLY this at the end of your message:\n\n"
        "<tool_calls>\n"
        '[{"name": "TOOL_NAME", "args": {"ARGUMENT": "VALUE"}}]\n'
        "</tool_calls>\n\n"
        "STEP 2 — You will receive a [Tool Result] message with the output.\n\n"
        "STEP 3 — Use the result to continue. Call more tools or give your final answer.\n\n"
        "═══ RULES ═══\n"
        "• Put <tool_calls> at the VERY END of your message — nothing after it.\n"
        "• Use ONLY the tools listed below. Do not invent tool names.\n"
        "• Arguments must be a JSON object with double-quoted keys and values.\n"
        "• Do NOT write fake results like 'The file contains...' — actually call the tool.\n"
        "• Do NOT say 'I will call...' or 'Let me use...' — just output the block.\n\n"
        f"═══ EXAMPLE ═══\n"
        f"User: Read the file app.py\n"
        f"You output:\n"
        f"<tool_calls>\n"
        f'[{{"name": "{first_tool}", "args": {{"path": "app.py"}}}}]\n'
        f"</tool_calls>\n\n"
        f"Then you receive:\n"
        f"[Tool Result]\n"
        f"# app.py\n"
        f"print('hello')\n\n"
        f"Then you reply: The file contains a single print statement.\n\n"
        "═══ AVAILABLE TOOLS ═══\n\n"
        + "\n\n".join(tool_specs)
    )


def _build_tool_system_reasoning(tools: list) -> str:
    """
    Compact prompt for reasoning/thinking models (gpt-oss-120b, qwen3-30b).
    Short so the model doesn't over-deliberate; explicit note to emit the
    block in response text, not in internal reasoning.
    """
    tool_specs = []
    for t in tools:
        fn       = t.get("function", t)
        name     = fn.get("name", "")
        desc     = fn.get("description", "")
        params   = fn.get("parameters", {})
        props    = params.get("properties", {})
        required = params.get("required", [])
        args_doc = []
        for k, v in props.items():
            req   = " (required)" if k in required else ""
            typ   = v.get("type", "any")
            pdesc = v.get("description", "")
            args_doc.append(f"  {k} ({typ}){req}: {pdesc}")
        tool_specs.append(
            f"Tool: {name}\n"
            f"Description: {desc}\n"
            "Arguments:\n" + ("\n".join(args_doc) if args_doc else "  (none)")
        )

    return (
        "\n\n---\n"
        "## TOOLS — EMIT IN RESPONSE TEXT (not in thinking)\n\n"
        "To call a tool, your RESPONSE TEXT must end with:\n"
        "<tool_calls>\n"
        '[{"name": "TOOL_NAME", "args": {"key": "value"}}]\n'
        "</tool_calls>\n\n"
        "RULES: Call immediately — no preamble. One block per turn. "
        "Multiple calls = JSON array. After [Tool Result] continues work.\n\n"
        "AVAILABLE TOOLS:\n\n"
        + "\n\n".join(tool_specs)
    )


def _build_tool_system(model_id: str, tools: list) -> str:
    if model_id in _REASONING_MODELS:
        return _build_tool_system_reasoning(tools)
    return _build_tool_system_small(tools)


# ── Message preprocessing ──────────────────────────────────────────────

def _preprocess_messages(messages: list) -> list:
    """
    Convert OpenAI tool-call history to text-based conversation.

    DeepAI models don't understand role:tool or assistant tool_calls.
    Convert them so the model sees a plain user/assistant chat.
    """
    out = []
    i = 0
    while i < len(messages):
        msg  = messages[i]
        role = msg.get("role", "")

        if role == "assistant" and msg.get("tool_calls"):
            content = msg.get("content", "") or ""
            tcs = msg["tool_calls"]
            tc_array = []
            for tc in tcs:
                fn   = tc.get("function", {})
                name = fn.get("name", "")
                try:
                    args = json.loads(fn.get("arguments", "{}"))
                except Exception:
                    args = {}
                tc_array.append({"name": name, "args": args})
            tc_block = (
                "\n<tool_calls>\n"
                + json.dumps(tc_array, ensure_ascii=False)
                + "\n</tool_calls>"
            )
            out.append({"role": "assistant", "content": content + tc_block})

        elif role == "tool":
            results = []
            while i < len(messages) and messages[i].get("role") == "tool":
                results.append(messages[i].get("content", ""))
                i += 1
            if len(results) == 1:
                body = f"[Tool Result]\n{results[0]}"
            else:
                body = "\n\n".join(
                    f"[Tool Result {j + 1}]\n{r}" for j, r in enumerate(results)
                )
            out.append({"role": "user", "content": body})
            continue

        else:
            out.append(msg)

        i += 1

    return out


def _inject_tool_system(model_id: str, messages: list, tools: list) -> list:
    ti   = _build_tool_system(model_id, tools)
    msgs = list(messages)
    if msgs and msgs[0].get("role") == "system":
        msgs[0] = dict(msgs[0])
        msgs[0]["content"] = msgs[0].get("content", "") + ti
    else:
        msgs.insert(0, {"role": "system", "content": ti.strip()})
    return msgs


# ── API call ───────────────────────────────────────────────────────────

def _call(model_id: str, messages: list[dict]) -> tuple[int, str]:
    ch = json.dumps([{"role": m["role"], "content": m.get("content", "")} for m in messages])
    try:
        r = requests.post(
            URL,
            data={"chat_style": "chat", "chatHistory": ch, "model": model_id, "enabled_tools": "[]"},
            headers={"api-key": API_KEY},
            timeout=120,
        )
        return r.status_code, r.text
    except Exception as e:
        return 0, str(e)


def _parse_tool_calls(text: str) -> tuple[str, list]:
    """Extract <tool_calls> block from response text."""
    if _TOOL_CALL_OPEN not in text:
        return text, []
    before, _, after = text.partition(_TOOL_CALL_OPEN)
    raw = after.split(_TOOL_CALL_CLOSE)[0].strip()
    try:
        tc = json.loads(raw)
        # Handle single object (model forgot the array wrapper)
        if isinstance(tc, dict):
            tc = [tc]
        if not isinstance(tc, list):
            tc = []
    except json.JSONDecodeError:
        tc = []
    return before.strip(), tc


def stream_chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> Generator[dict, None, None]:
    messages = _preprocess_messages(messages)
    if tools:
        messages = _inject_tool_system(model_id, messages, tools)

    status, text = _call(model_id, messages)

    if status == 401:
        yield {"type": "error", "text": "Pro model — not available on free tier"}
        return
    if status != 200:
        yield {"type": "error", "text": f"HTTP {status}: {text[:200]}"}
        return

    text, tcs = _parse_tool_calls(text)

    # Stream text in chunks
    chunk_size = 20
    for i in range(0, len(text), chunk_size):
        yield {"type": "text", "text": text[i:i + chunk_size]}

    # Emit tool_delta events
    for i, tc in enumerate(tcs):
        name = tc.get("name", "")
        args = tc.get("args", tc.get("arguments", {}))
        yield {
            "type":      "tool_delta",
            "index":     i,
            "id":        "",
            "name":      name,
            "arguments": json.dumps(args) if not isinstance(args, str) else args,
        }

    yield {"type": "done"}


def chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> dict:
    messages = _preprocess_messages(messages)
    if tools:
        messages = _inject_tool_system(model_id, messages, tools)

    status, text = _call(model_id, messages)

    if status == 401:
        return {"type": "error", "text": "Pro model — not available on free tier"}
    if status != 200:
        return {"type": "error", "text": f"HTTP {status}: {text[:200]}"}

    text, tcs = _parse_tool_calls(text)
    return {"type": "text", "text": text, "tool_calls": tcs}
