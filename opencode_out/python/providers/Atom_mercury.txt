"""
Free provider — MercuryAI (chat.mercuryai.io) via direct API.
No authentication required - works for guest/anonymous users.
Endpoint: https://chat.mercuryai.io/api/chat

Tool calling uses text-based scheme (like qwen provider):
model outputs <tool_calls>[...]</tool_calls> at end of response.
"""

import json
import uuid
import requests
from typing import Generator

PROVIDER_NAME = "Atom mercury"
MODELS = [
    {"id": "mercury-atom", "label": "Mercury Atom", "ctx": 128000},
    {"id": "cohere-command-r-plus-08", "label": "Cohere Command R+ 08", "ctx": 128000},
    {"id": "cohere-command-a", "label": "Cohere Command A", "ctx": 128000},
]

BASE_URL = "https://chat.mercuryai.io"
CHAT_ENDPOINT = f"{BASE_URL}/api/chat"

_TOOL_CALL_OPEN = "<tool_calls>"
_TOOL_CALL_CLOSE = "</tool_calls>"


def _build_messages(messages: list[dict]) -> list[dict]:
    formatted = []
    for msg in messages:
        formatted.append({
            "id": str(uuid.uuid4())[:12],
            "role": msg.get("role", "user"),
            "content": msg.get("content", ""),
            "createdAt": msg.get("createdAt", ""),
            "attachments": msg.get("attachments", [])
        })
    return formatted


def _build_tool_system(tools: list) -> str:
    """Build tool instructions for text-based tool calling."""
    tool_specs = []
    for t in tools:
        fn = t.get("function", t)
        name = fn.get("name", "")
        desc = fn.get("description", "")
        params = fn.get("parameters", {})
        props = params.get("properties", {})
        required = params.get("required", [])
        args_doc = []
        for k, v in props.items():
            req = " (required)" if k in required else ""
            typ = v.get("type", "any")
            pdesc = v.get("description", "")
            args_doc.append(f"  - {k} ({typ}){req}: {pdesc}")
        tool_specs.append(
            f"### {name}\n{desc}\nArguments:\n" + ("\n".join(args_doc) if args_doc else "  (none)")
        )

    return (
        "\n\n---\n"
        "## Tool Calling\n\n"
        "You have access to tools. If you need to call one or more tools, "
        "output your full response text first, then append a tool call block "
        "as the VERY LAST thing in your message.\n\n"
        "Format:\n"
        "<tool_calls>\n"
        '[{"name": "tool_name", "args": {"arg1": "val1"}}, ...]\n'
        "</tool_calls>\n\n"
        "Rules:\n"
        "- Only include the block if you are actually calling a tool.\n"
        "- All tool calls for a single turn go in one block as a JSON array.\n"
        "- After you receive tool results continue your response.\n\n"
        "Available tools:\n\n" + "\n\n".join(tool_specs)
    )


def _inject_tool_system(messages: list, tools: list) -> list:
    """Inject tool instructions into messages."""
    tool_instructions = _build_tool_system(tools)
    msgs = list(messages)
    if msgs and msgs[0].get("role") == "system":
        msgs[0] = dict(msgs[0])
        msgs[0]["content"] = msgs[0].get("content", "") + "\n" + tool_instructions
    else:
        msgs.insert(0, {"role": "system", "content": tool_instructions.strip()})
    return msgs


def stream_chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> Generator[dict, None, None]:
    """Stream chat via MercuryAI with text-based tool calling."""
    url = CHAT_ENDPOINT
    conversation_id = str(uuid.uuid4())

    # Inject tool instructions if tools provided
    if tools:
        messages = _inject_tool_system(messages, tools)

    payload = {
        "messages": _build_messages(messages),
        "conversationId": conversation_id,
        "attachments": [],
        "modelId": model_id,
        "useSearch": False
    }

    headers = {
        "Content-Type": "application/json",
        "Origin": BASE_URL,
        "Referer": f"{BASE_URL}/"
    }

    try:
        response = requests.post(url, json=payload, headers=headers, timeout=120)
        response.raise_for_status()
    except Exception as e:
        yield {"type": "error", "text": str(e)}
        return

    try:
        data = response.json()
    except json.JSONDecodeError as e:
        yield {"type": "error", "text": f"Failed to parse response: {e}"}
        return

    if "error" in data:
        yield {"type": "error", "text": data.get("error", "Unknown error")}
        return

    response_text = data.get("response", "")

    # Check for tool_calls in response
    tool_buffer = None
    if _TOOL_CALL_OPEN in response_text:
        before, _, after = response_text.partition(_TOOL_CALL_OPEN)
        if before:
            yield {"type": "text", "text": before}
        tool_buffer = after.split(_TOOL_CALL_CLOSE)[0].strip()
    else:
        # No tools - yield entire response as chunks
        chunk_size = 20
        for i in range(0, len(response_text), chunk_size):
            yield {"type": "text", "text": response_text[i:i + chunk_size]}

    # Parse and yield tool calls
    if tool_buffer:
        try:
            tool_calls = json.loads(tool_buffer)
            for i, tc in enumerate(tool_calls):
                name = tc.get("name", "")
                args = tc.get("args", tc.get("arguments", {}))
                yield {
                    "type": "tool_delta",
                    "index": i,
                    "id": "",
                    "name": name,
                    "arguments": json.dumps(args) if not isinstance(args, str) else args,
                }
        except json.JSONDecodeError as e:
            yield {"type": "error", "text": f"Failed to parse tool_calls: {e}"}

    yield {"type": "done"}


def chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> dict:
    """Non-streaming chat via MercuryAI with text-based tool calling."""
    url = CHAT_ENDPOINT
    conversation_id = str(uuid.uuid4())

    # Inject tool instructions if tools provided
    if tools:
        messages = _inject_tool_system(messages, tools)

    payload = {
        "messages": _build_messages(messages),
        "conversationId": conversation_id,
        "attachments": [],
        "modelId": model_id,
        "useSearch": False
    }

    headers = {
        "Content-Type": "application/json",
        "Origin": BASE_URL,
        "Referer": f"{BASE_URL}/"
    }

    try:
        response = requests.post(url, json=payload, headers=headers, timeout=120)
        response.raise_for_status()
    except Exception as e:
        return {"type": "error", "text": str(e)}

    try:
        data = response.json()
    except json.JSONDecodeError:
        return {"type": "error", "text": "Failed to parse response"}

    if "error" in data:
        return {"type": "error", "text": data.get("error", "Unknown error")}

    response_text = data.get("response", "")
    tool_calls = []

    # Check for tool_calls in response
    if _TOOL_CALL_OPEN in response_text:
        before, _, after = response_text.partition(_TOOL_CALL_OPEN)
        response_text = before.strip()
        raw_block = after.split(_TOOL_CALL_CLOSE)[0].strip()
        try:
            tool_calls = json.loads(raw_block)
            if not isinstance(tool_calls, list):
                tool_calls = []
        except json.JSONDecodeError:
            tool_calls = []

    return {
        "type": "text",
        "text": response_text,
        "tool_calls": tool_calls,
    }
