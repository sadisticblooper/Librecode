"""
Free provider — MercuryAI (chat.mercuryai.io) via direct API.
No authentication required — works for guest/anonymous users.
Endpoint: https://chat.mercuryai.io/api/chat

Tool calling is text-based (model appends a <tool_calls> JSON block).

FIXES vs previous version:
- _preprocess_messages() converts role:tool → role:user so multi-turn
  tool use works: model can see previous results and not re-call the same tool.
- _build_tool_system() rewritten to be more imperative with an example.
"""

import json
import uuid
import requests
from typing import Generator

PROVIDER_NAME = "Atom mercury"
MODELS = [
    {"id": "mercury-atom",              "label": "Mercury Atom",       "ctx": 128000},
    {"id": "cohere-command-r-plus-08",  "label": "Cohere Command R+ 08", "ctx": 128000},
    {"id": "cohere-command-a",          "label": "Cohere Command A",   "ctx": 128000},
]

BASE_URL      = "https://chat.mercuryai.io"
CHAT_ENDPOINT = f"{BASE_URL}/api/chat"

_TOOL_CALL_OPEN  = "<tool_calls>"
_TOOL_CALL_CLOSE = "</tool_calls>"


# ── Tool prompt ────────────────────────────────────────────────────────

def _build_tool_system(tools: list) -> str:
    tool_specs = []
    for t in tools:
        fn       = t.get("function", t)
        name     = fn.get("name", "")
        desc     = fn.get("description", "")
        params   = fn.get("parameters", {})
        props    = params.get("properties", {})
        required = params.get("required", [])
        args_doc = []
        for k, v in props.items():
            req   = " (required)" if k in required else ""
            typ   = v.get("type", "any")
            pdesc = v.get("description", "")
            args_doc.append(f"  {k} ({typ}){req}: {pdesc}")
        tool_specs.append(
            f"Tool: {name}\n"
            f"Description: {desc}\n"
            "Arguments:\n" + ("\n".join(args_doc) if args_doc else "  (none)")
        )

    first_tool = tools[0].get("function", tools[0]).get("name", "some_tool") if tools else "some_tool"

    return (
        "\n\n---\n"
        "## TOOL USE\n\n"
        "You have tools. When a task requires a tool, call it IMMEDIATELY — "
        "do NOT describe what you plan to do, just output the <tool_calls> block.\n\n"
        "FORMAT (valid JSON, double quotes, at the VERY END of your message):\n"
        "<tool_calls>\n"
        '[{"name": "TOOL_NAME", "args": {"key": "value"}}]\n'
        "</tool_calls>\n\n"
        "RULES:\n"
        "1. Call tools without preamble — no 'I will...' or 'Let me...'.\n"
        "2. One <tool_calls> block per message; multiple calls go in the JSON array.\n"
        "3. After you receive a [Tool Result] message, continue your work.\n"
        "4. NEVER simulate or guess tool outputs.\n\n"
        f"EXAMPLE:\n"
        f"User: Read README.md\n"
        f"You output:\n"
        f"<tool_calls>\n"
        f'[{{"name": "{first_tool}", "args": {{"path": "README.md"}}}}]\n'
        f"</tool_calls>\n\n"
        "AVAILABLE TOOLS:\n\n"
        + "\n\n".join(tool_specs)
    )


# ── Message preprocessing ──────────────────────────────────────────────

def _preprocess_messages(messages: list) -> list:
    """
    Convert OpenAI tool-call history to text-based conversation.

    Converts:
      assistant with tool_calls list  →  assistant with <tool_calls> in content
      role:tool messages              →  role:user [Tool Result] messages
    """
    out = []
    i = 0
    while i < len(messages):
        msg  = messages[i]
        role = msg.get("role", "")

        if role == "assistant" and msg.get("tool_calls"):
            content = msg.get("content", "") or ""
            tcs = msg["tool_calls"]
            tc_array = []
            for tc in tcs:
                fn   = tc.get("function", {})
                name = fn.get("name", "")
                try:
                    args = json.loads(fn.get("arguments", "{}"))
                except Exception:
                    args = {}
                tc_array.append({"name": name, "args": args})
            tc_block = (
                "\n<tool_calls>\n"
                + json.dumps(tc_array, ensure_ascii=False)
                + "\n</tool_calls>"
            )
            out.append({"role": "assistant", "content": content + tc_block})

        elif role == "tool":
            results = []
            while i < len(messages) and messages[i].get("role") == "tool":
                results.append(messages[i].get("content", ""))
                i += 1
            if len(results) == 1:
                body = f"[Tool Result]\n{results[0]}"
            else:
                body = "\n\n".join(
                    f"[Tool Result {j + 1}]\n{r}" for j, r in enumerate(results)
                )
            out.append({"role": "user", "content": body})
            continue

        else:
            out.append(msg)

        i += 1

    return out


def _inject_tool_system(messages: list, tools: list) -> list:
    tool_instructions = _build_tool_system(tools)
    msgs = list(messages)
    if msgs and msgs[0].get("role") == "system":
        msgs[0] = dict(msgs[0])
        msgs[0]["content"] = msgs[0].get("content", "") + tool_instructions
    else:
        msgs.insert(0, {"role": "system", "content": tool_instructions.strip()})
    return msgs


# ── MercuryAI message builder ──────────────────────────────────────────

def _build_messages(messages: list[dict]) -> list[dict]:
    formatted = []
    for msg in messages:
        formatted.append({
            "id":          str(uuid.uuid4())[:12],
            "role":        msg.get("role", "user"),
            "content":     msg.get("content", ""),
            "createdAt":   msg.get("createdAt", ""),
            "attachments": msg.get("attachments", []),
        })
    return formatted


# ── Tool call parser ───────────────────────────────────────────────────

def _parse_tool_calls(text: str) -> tuple[str, list]:
    if _TOOL_CALL_OPEN not in text:
        return text, []
    before, _, after = text.partition(_TOOL_CALL_OPEN)
    raw = after.split(_TOOL_CALL_CLOSE)[0].strip()
    try:
        tc = json.loads(raw)
        if isinstance(tc, dict):
            tc = [tc]
        if not isinstance(tc, list):
            tc = []
    except json.JSONDecodeError:
        tc = []
    return before.strip(), tc


# ── API call ───────────────────────────────────────────────────────────

def stream_chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> Generator[dict, None, None]:
    messages = _preprocess_messages(messages)
    if tools:
        messages = _inject_tool_system(messages, tools)

    payload = {
        "messages":       _build_messages(messages),
        "conversationId": str(uuid.uuid4()),
        "attachments":    [],
        "modelId":        model_id,
        "useSearch":      False,
    }
    headers = {
        "Content-Type": "application/json",
        "Origin":        BASE_URL,
        "Referer":       f"{BASE_URL}/",
    }

    try:
        response = requests.post(CHAT_ENDPOINT, json=payload, headers=headers, timeout=120)
        response.raise_for_status()
    except Exception as e:
        yield {"type": "error", "text": str(e)}
        return

    try:
        data = response.json()
    except json.JSONDecodeError as e:
        yield {"type": "error", "text": f"Failed to parse response: {e}"}
        return

    if "error" in data:
        yield {"type": "error", "text": data.get("error", "Unknown error")}
        return

    response_text = data.get("response", "")
    text, tcs     = _parse_tool_calls(response_text)

    # Stream text
    chunk_size = 20
    for i in range(0, len(text), chunk_size):
        yield {"type": "text", "text": text[i:i + chunk_size]}

    # Emit tool_delta events
    for i, tc in enumerate(tcs):
        name = tc.get("name", "")
        args = tc.get("args", tc.get("arguments", {}))
        yield {
            "type":      "tool_delta",
            "index":     i,
            "id":        "",
            "name":      name,
            "arguments": json.dumps(args) if not isinstance(args, str) else args,
        }

    yield {"type": "done"}


def chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> dict:
    messages = _preprocess_messages(messages)
    if tools:
        messages = _inject_tool_system(messages, tools)

    payload = {
        "messages":       _build_messages(messages),
        "conversationId": str(uuid.uuid4()),
        "attachments":    [],
        "modelId":        model_id,
        "useSearch":      False,
    }
    headers = {
        "Content-Type": "application/json",
        "Origin":        BASE_URL,
        "Referer":       f"{BASE_URL}/",
    }

    try:
        response = requests.post(CHAT_ENDPOINT, json=payload, headers=headers, timeout=120)
        response.raise_for_status()
    except Exception as e:
        return {"type": "error", "text": str(e)}

    try:
        data = response.json()
    except json.JSONDecodeError:
        return {"type": "error", "text": "Failed to parse response"}

    if "error" in data:
        return {"type": "error", "text": data.get("error", "Unknown error")}

    response_text = data.get("response", "")
    text, tcs     = _parse_tool_calls(response_text)

    return {"type": "text", "text": text, "tool_calls": tcs}
