"""
Free provider — qwen2api (chat.qwen.ai) via smanx's public proxy.
Proxy: https://github.com/smanx/qwen2api
Endpoint: https://qwen2api-n.smanx.xx.kg/v1/chat/completions

Tool calling is text-based: model appends a <tool_calls> JSON block at end
of any message that needs tools.

FIXES vs previous version:
- _preprocess_messages() converts role:tool results → role:user so Qwen
  can actually SEE the results and stop re-calling the same tool.
- _build_tool_system() rewritten to be more imperative: call immediately,
  no preamble, concrete example included.
"""

import os
import requests
from typing import Generator

PROVIDER_NAME = "qwen"
MODELS = [
    {"id": "qwen3.6-plus",               "label": "Qwen 3.6 Plus",        "ctx": 1000000},
    {"id": "qwen3.6-max-preview",        "label": "Qwen 3.6 Max Preview", "ctx": 262144},
    {"id": "qwen3.6-27b",                "label": "Qwen 3.6 27B",         "ctx": 262144},
    {"id": "qwen3.5-plus",               "label": "Qwen 3.5 Plus",        "ctx": 128000},
    {"id": "qwen3.5-flash",              "label": "Qwen 3.5 Flash",       "ctx": 128000},
    {"id": "qwen3.5-turbo",              "label": "Qwen 3.5 Turbo",       "ctx": 128000},
    {"id": "qwen2.5-coder-32b-instruct", "label": "Qwen 2.5 Coder 32B",  "ctx": 128000},
]

BASE_URL = os.environ.get("QWEN_PROXY_URL", "https://qwen2api-n.smanx.xx.kg")

_TOOL_CALL_OPEN  = "<tool_calls>"
_TOOL_CALL_CLOSE = "</tool_calls>"
_MARKER_LEN      = len(_TOOL_CALL_OPEN)


# ── Tool prompt ────────────────────────────────────────────────────────

def _build_tool_system(tools: list) -> str:
    """
    Build tool instructions. Imperative tone: call immediately, no preamble.
    Includes a concrete example so the model can pattern-match.
    """
    tool_specs = []
    for t in tools:
        fn       = t.get("function", t)
        name     = fn.get("name", "")
        desc     = fn.get("description", "")
        params   = fn.get("parameters", {})
        props    = params.get("properties", {})
        required = params.get("required", [])
        args_doc = []
        for k, v in props.items():
            req   = " (required)" if k in required else ""
            typ   = v.get("type", "any")
            pdesc = v.get("description", "")
            args_doc.append(f"  {k} ({typ}){req}: {pdesc}")
        tool_specs.append(
            f"Tool: {name}\n"
            f"Description: {desc}\n"
            "Arguments:\n" + ("\n".join(args_doc) if args_doc else "  (none)")
        )

    return (
        "\n\n---\n"
        "## TOOL USE\n\n"
        "You have tools. When a task requires information from a file, terminal, "
        "or any tool — call it IMMEDIATELY. Do NOT say 'I will read...' or describe "
        "what you plan to do. Just output the <tool_calls> block.\n\n"
        "FORMAT (valid JSON, double quotes, at the VERY END of your message):\n"
        "<tool_calls>\n"
        '[{"name": "TOOL_NAME", "args": {"key": "value"}}]\n'
        "</tool_calls>\n\n"
        "RULES:\n"
        "1. Output the block immediately — no explanatory text before it.\n"
        "2. One block per turn; put multiple calls in the JSON array.\n"
        "3. After you receive results (they arrive as a [Tool Result] message), "
        "continue your work — call more tools or write your final answer.\n"
        "4. NEVER simulate or guess tool results.\n\n"
        "EXAMPLE — user says: edit main.py to add a logger\n"
        "  CORRECT response:\n"
        "  <tool_calls>\n"
        '  [{"name": "read_file", "args": {"path": "main.py"}}]\n'
        "  </tool_calls>\n"
        "  WRONG response:\n"
        "  \"Let me first read main.py to understand its structure.\"\n\n"
        "AVAILABLE TOOLS:\n\n"
        + "\n\n".join(tool_specs)
    )


# ── Message preprocessing ──────────────────────────────────────────────

def _preprocess_messages(messages: list) -> list:
    """
    Convert OpenAI tool-call history to text-based conversation.

    routes.py stores tool results as role:tool messages and assistant
    tool_calls as a list. The Qwen proxy doesn't understand either.
    This function converts them so the model sees a coherent history:

      assistant (with tool_calls)  →  assistant with <tool_calls> in content
      role:tool                    →  role:user [Tool Result]
    """
    out = []
    i = 0
    while i < len(messages):
        msg  = messages[i]
        role = msg.get("role", "")

        if role == "assistant" and msg.get("tool_calls"):
            # Rebuild assistant message with embedded <tool_calls> block
            content = msg.get("content", "") or ""
            tcs = msg["tool_calls"]
            tc_array = []
            for tc in tcs:
                fn   = tc.get("function", {})
                name = fn.get("name", "")
                try:
                    args = json.loads(fn.get("arguments", "{}"))
                except Exception:
                    args = {}
                tc_array.append({"name": name, "args": args})
            tc_block = (
                "\n<tool_calls>\n"
                + json.dumps(tc_array, ensure_ascii=False)
                + "\n</tool_calls>"
            )
            out.append({"role": "assistant", "content": content + tc_block})

        elif role == "tool":
            # Collect consecutive tool results → single user message
            results = []
            while i < len(messages) and messages[i].get("role") == "tool":
                results.append(messages[i].get("content", ""))
                i += 1
            if len(results) == 1:
                body = f"[Tool Result]\n{results[0]}"
            else:
                body = "\n\n".join(
                    f"[Tool Result {j + 1}]\n{r}" for j, r in enumerate(results)
                )
            out.append({"role": "user", "content": body})
            continue  # i already advanced

        else:
            out.append(msg)

        i += 1

    return out


def _inject_tool_system(messages: list, tools: list) -> list:
    """Append tool instructions to the system message."""
    tool_instructions = _build_tool_system(tools)
    msgs = list(messages)
    if msgs and msgs[0].get("role") == "system":
        msgs[0] = dict(msgs[0])
        msgs[0]["content"] = msgs[0].get("content", "") + tool_instructions
    else:
        msgs.insert(0, {"role": "system", "content": tool_instructions.strip()})
    return msgs


# ── API call ───────────────────────────────────────────────────────────

def stream_chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> Generator[dict, None, None]:
    url = f"{BASE_URL}/v1/chat/completions"

    # Always preprocess: converts role:tool → role:user so model sees results
    messages = _preprocess_messages(messages)
    if tools:
        messages = _inject_tool_system(messages, tools)

    payload = {
        "model":      model_id,
        "messages":   messages,
        "stream":     True,
        "max_tokens": max_tokens or 4096,
    }

    try:
        api_resp = requests.post(url, json=payload, stream=True, timeout=(15, 120))
        api_resp.raise_for_status()
    except Exception as e:
        yield {"type": "error", "text": str(e)}
        return

    pending     = ""
    tool_buffer = None

    try:
        for raw in api_resp.iter_lines():
            if not raw:
                continue
            line = raw.decode("utf-8", errors="replace")
            if not line.startswith("data: "):
                continue
            data_str = line[6:].strip()
            if data_str == "[DONE]":
                break
            try:
                chunk = json.loads(data_str)
            except json.JSONDecodeError:
                continue

            choice = (chunk.get("choices") or [{}])[0]
            delta  = choice.get("delta", {})

            thinking = (
                delta.get("reasoning_content")
                or delta.get("reasoning")
                or delta.get("thinking")
                or ""
            )
            if thinking:
                yield {"type": "thinking", "text": thinking}

            content = delta.get("content") or ""
            if not content:
                continue

            if tool_buffer is not None:
                tool_buffer += content
                continue

            pending += content

            if _TOOL_CALL_OPEN in pending:
                before, _, after = pending.partition(_TOOL_CALL_OPEN)
                if before:
                    yield {"type": "text", "text": before}
                tool_buffer = after
                pending     = ""
            else:
                safe_len = max(0, len(pending) - (_MARKER_LEN - 1))
                if safe_len > 0:
                    yield {"type": "text", "text": pending[:safe_len]}
                    pending = pending[safe_len:]

    except Exception as e:
        yield {"type": "error", "text": f"Stream interrupted: {e}"}
        return

    if tool_buffer is None:
        if pending:
            yield {"type": "text", "text": pending}
        yield {"type": "done"}
        return

    raw_block = (tool_buffer + pending).split(_TOOL_CALL_CLOSE)[0].strip()
    try:
        tool_calls = json.loads(raw_block)
        if not isinstance(tool_calls, list):
            raise ValueError("expected JSON array")
    except Exception as e:
        yield {"type": "error", "text": f"Failed to parse tool_calls: {e}\nRaw: {raw_block}"}
        yield {"type": "done"}
        return

    for i, tc in enumerate(tool_calls):
        name = tc.get("name", "")
        args = tc.get("args", tc.get("arguments", {}))
        yield {
            "type":      "tool_delta",
            "index":     i,
            "id":        "",
            "name":      name,
            "arguments": json.dumps(args) if not isinstance(args, str) else args,
        }

    yield {"type": "done"}


def chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> dict:
    url = f"{BASE_URL}/v1/chat/completions"

    messages = _preprocess_messages(messages)
    if tools:
        messages = _inject_tool_system(messages, tools)

    payload = {
        "model":      model_id,
        "messages":   messages,
        "stream":     False,
        "max_tokens": max_tokens or 4096,
    }

    try:
        resp = requests.post(url, json=payload, timeout=600)
        resp.raise_for_status()
        data    = resp.json()
        choice  = (data.get("choices") or [{}])[0]
        msg     = choice.get("message", {})
        content = msg.get("content", "")

        tool_calls = []
        if tools and _TOOL_CALL_OPEN in content:
            before, _, after = content.partition(_TOOL_CALL_OPEN)
            raw_block = after.split(_TOOL_CALL_CLOSE)[0].strip()
            content   = before.strip()
            try:
                parsed = json.loads(raw_block)
                if isinstance(parsed, list):
                    tool_calls = parsed
            except Exception:
                pass

        return {
            "type":       "text",
            "text":       content,
            "thinking":   msg.get("reasoning_content", ""),
            "tool_calls": tool_calls,
        }
    except Exception as e:
        return {"type": "error", "text": str(e)}
