def _prepare_messages(messages: list[dict]) -> list[dict]:
    """
    Convert messages from OpenAI format (used internally by routes.py)
    to Ollama's native /api/chat format:
    - tool_calls arguments: JSON string → dict
    - tool result messages: drop tool_call_id (Ollama doesn't use it)
    """
    out = []
    for msg in messages:
        role = msg.get("role")

        if role == "assistant" and msg.get("tool_calls"):
            fixed_tcs = []
            for tc in msg["tool_calls"]:
                func = tc.get("function", {})
                args = func.get("arguments", {})
                # routes.py stores arguments as a JSON string — convert to dict
                if isinstance(args, str):
                    try:
                        args = json.loads(args)
                    except (json.JSONDecodeError, ValueError):
                        args = {}
                fixed_tcs.append({
                    "function": {
                        "name":      func.get("name", ""),
                        "arguments": args,
                    }
                })
            out.append({
                "role":       "assistant",
                "content":    msg.get("content") or "",
                "tool_calls": fixed_tcs,
            })

        elif role == "tool":
            # Ollama tool results: just role + content, no tool_call_id
            out.append({
                "role":    "tool",
                "content": msg.get("content", ""),
            })

        else:
            out.append(msg)

    return out


def stream_chat(model_id: str, messages: list[dict], tools: list | None = None,
    _ensure_models()
                max_tokens: int | None = None) -> Generator[dict, None, None]:
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type":  "application/json",
    }
    payload = {
        "model":    model_id,
        "messages": _prepare_messages(messages),
        "stream":   True,
    }
    if max_tokens:
        payload["options"] = {"num_predict": max_tokens}
    if tools:
        payload["tools"] = tools

    try:
        resp = requests.post(
            f"{API_BASE}/api/chat",
            headers=headers,
            json=payload,
            stream=True,
            timeout=600,
        )
        resp.raise_for_status()
    except Exception as e:
        yield {"type": "error", "text": str(e)}
        return

    for line in resp.iter_lines(decode_unicode=True):
        if not line:
            continue
        try:
            chunk = json.loads(line)
        except json.JSONDecodeError:
            continue

        msg = chunk.get("message", {})

        thinking = msg.get("reasoning_content") or msg.get("thinking") or ""
        if thinking:
            yield {"type": "thinking", "text": thinking}

        content = msg.get("content") or ""
        if content:
            yield {"type": "text", "text": content}

        # tool calls — Ollama returns arguments as dict, convert to JSON string
        # for routes.py which expects the OpenAI format
        for tc in msg.get("tool_calls", []):
            func = tc.get("function", {})
            args = func.get("arguments", {})
            yield {
                "type":      "tool_delta",
                "index":     0,
                "id":        tc.get("id", ""),
                "name":      func.get("name", ""),
                "arguments": json.dumps(args) if isinstance(args, dict) else (args or ""),
            }

        if chunk.get("done"):
            yield {"type": "done"}
            return
