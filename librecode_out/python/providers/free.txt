"""
Free provider — opencode.ai Zen API via SSE.

Provider capability flags (read by providers/__init__.py):

  MODEL_EFFORT             — reasoning_effort value to send per model.
                             Models absent from this dict don't get the field at all.

  NEEDS_REASONING_PASSBACK — model IDs that require reasoning_content echoed back
                             in every assistant message during multi-turn (DeepSeek V4).

"""
import json
import uuid
import requests
from typing import Generator

PROVIDER_NAME = "free"
MODELS = [
    {"id": "deepseek-v4-flash-free",  "label": "DeepSeek V4 Flash",  "ctx": 128000},
    {"id": "mimo-v2.5-free",          "label": "MiMo V2.5",          "ctx": 128000},
    {"id": "minimax-m3-free",         "label": "MiniMax M3",         "ctx": 128000},
    {"id": "nemotron-3-super-free",   "label": "Nemotron 3 Super",   "ctx": 128000},
    {"id": "big-pickle",              "label": "Big Pickle",         "ctx": 128000},
]

# reasoning_effort to send per model.
# minimax-m3 accepts the field but ignores it — send medium as a safe default.
# nemotron: "max" breaks it (probe confirmed), "xhigh" is the ceiling that works.
# mimo: uses "reasoning" delta field (not reasoning_content), effort accepted, no passback.
MODEL_EFFORT = {
    "deepseek-v4-flash-free": "max",
    "mimo-v2.5-free":         "high",  # "max" unverified; "high" confirmed working
    "nemotron-3-super-free":  "xhigh",
    "big-pickle":             "max",
}

# All supported reasoning_effort levels per model (for UI pickers etc.)
# nemotron: "max" confirmed broken by probe — excluded.
# mimo: same effort range as deepseek (reasoning-first model, same API behaviour).
# minimax-m3: field accepted but no reasoning field returned — all levels listed
#             for completeness but effectively cosmetic.
MODEL_LEVELS = {
    "deepseek-v4-flash-free": ["low", "medium", "high", "max"],
    "mimo-v2.5-free":         ["low", "medium", "high", "max"],
    "minimax-m3-free":        ["none", "minimal", "low", "medium", "high", "xhigh", "max"],
    "nemotron-3-super-free":  ["none", "minimal", "low", "medium", "high", "xhigh"],
    "big-pickle":             ["low", "medium", "high", "xhigh", "max"],
}

# Models where thinking mode is always on and reasoning_content MUST be echoed
# back in every assistant message during multi-turn, or the API 400s.
# mimo uses the "thinking" delta field instead of "reasoning_content" but the
# same passback rule applies — include it defensively.
# deepseek-v4-flash-free: reasoning_content field, MUST be echoed back in multi-turn.
# mimo-v2.5-free: uses "reasoning" field (output-only), no echo needed — passback causes 400.
NEEDS_REASONING_PASSBACK = {"deepseek-v4-flash-free"}

# minimax-m3-free: reasoning lives inside <think>...</think> tags in content.
INLINE_THINK_MODELS = {"minimax-m3-free"}


# _parse_inline_thinking removed — replaced with stateful buffer in stream_chat


def stream_chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
    reasoning_effort: str | None = None,
) -> Generator[dict, None, None]:
    url        = "https://opencode.ai/zen/v1/chat/completions"
    session_id = f"ses_{uuid.uuid4().hex[:16]}"
    request_id = f"msg_{uuid.uuid4().hex[:16]}"

    headers = {
        "Authorization":      "Bearer public",
        "Content-Type":       "application/json",
        "User-Agent":         "opencode/1.15.3 ai-sdk/provider-utils/4.0.23 runtime/bun/1.3.13",
        "x-opencode-client":  "cli",
        "x-opencode-project": "global",
        "x-opencode-request": request_id,
        "x-opencode-session": session_id,
    }

    payload: dict = {
        "model":      model_id,
        "messages":   messages,
        "stream":     True,
        "max_tokens": max_tokens or 4096,
    }

    # Only send reasoning_effort for models that support it.
    effort = reasoning_effort or MODEL_EFFORT.get(model_id)
    if effort:
        payload["reasoning_effort"] = effort

    if tools:
        payload["tools"]       = tools
        payload["tool_choice"] = "auto"

    try:
        api_resp = requests.post(url, json=payload, headers=headers, stream=True, timeout=600)
        api_resp.raise_for_status()
    except Exception as e:
        yield {"type": "error", "text": str(e)}
        return

    inline_think = model_id in INLINE_THINK_MODELS
    # Stateful buffer for inline-think models (<think>…</think> split across chunks)
    _it_buf      = ""   # accumulated unprocessed content
    _it_thinking = False # False = in content mode; True = inside <think> block

    for raw in api_resp.iter_lines():
        if not raw:
            continue
        line = raw.decode("utf-8", errors="replace")

        if not line.startswith("data:"):
            continue
        ds = line[5:].strip()
        if not ds or ds == "[DONE]":
            yield {"type": "done"}
            return
        try:
            chunk = json.loads(ds)
        except json.JSONDecodeError:
            continue

        # OpenAI-style: choices[].delta
        choices = chunk.get("choices")
        if choices:
            delta = choices[0].get("delta", {})

            # Dedicated reasoning field (DeepSeek, etc.)
            thinking = (
                delta.get("reasoning_content")
                or delta.get("reasoning")
                or delta.get("thinking")
                or delta.get("reasoning_thought")
                or ""
            )
            if thinking:
                yield {"type": "thinking", "text": thinking}

            content = delta.get("content") or ""
            if content:
                if inline_think:
                    _it_buf += content
                    # Drain the buffer, honouring <think>…</think> boundaries
                    while _it_buf:
                        if _it_thinking:
                            if "</think>" in _it_buf:
                                thought, _, _it_buf = _it_buf.partition("</think>")
                                if thought:
                                    yield {"type": "thinking", "text": thought}
                                _it_thinking = False
                            else:
                                yield {"type": "thinking", "text": _it_buf}
                                _it_buf = ""
                        else:
                            if "<think>" in _it_buf:
                                pre, _, _it_buf = _it_buf.partition("<think>")
                                if pre:
                                    yield {"type": "text", "text": pre}
                                _it_thinking = True
                            else:
                                yield {"type": "text", "text": _it_buf}
                                _it_buf = ""
                else:
                    yield {"type": "text", "text": content}

            for tc_delta in delta.get("tool_calls", []):
                idx  = tc_delta.get("index", 0)
                func = tc_delta.get("function", {})
                yield {
                    "type":      "tool_delta",
                    "index":     idx,
                    "id":        tc_delta.get("id", ""),
                    "name":      func.get("name", ""),
                    "arguments": func.get("arguments", ""),
                }

        usage = chunk.get("usage")
        if usage:
            yield {
                "type":         "usage",
                "input_tokens":  usage.get("prompt_tokens", 0),
                "output_tokens": usage.get("completion_tokens", 0),
            }

        # Anthropic-style: delta.type == thinking_delta / text_delta
        delta = chunk.get("delta", {})
        dtype = delta.get("type", "")
        if dtype == "thinking_delta":
            yield {"type": "thinking", "text": delta.get("thinking", "")}
        elif dtype == "text_delta":
            yield {"type": "text", "text": delta.get("text", "")}
