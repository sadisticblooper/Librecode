"""
chatgpt_web.py — Provider that drives ChatGPT via the DSL scripting engine.

Mirrors demo.py exactly — calls the executor directly instead of
going through the HTTP /v1/chat/completions round-trip.
"""

import json
from typing import Generator

PROVIDER_NAME = "chatgpt_web"
SCRIPT_ID     = "chatgpt"

MODELS = [
    {"id": "chatgpt",    "label": "ChatGPT (Web)",    "ctx": 32000, "script_id": SCRIPT_ID},
    {"id": "chatgpt-4o", "label": "ChatGPT 4o (Web)", "ctx": 32000, "script_id": SCRIPT_ID},
]

_TOOL_CALL_OPEN  = "<tool_calls>"
_TOOL_CALL_CLOSE = "</tool_calls>"


def _build_tool_system(tools: list) -> str:
    tool_specs = []
    for t in tools:
        fn       = t.get("function", t)
        name     = fn.get("name", "")
        desc     = fn.get("description", "")
        params   = fn.get("parameters", {})
        props    = params.get("properties", {})
        required = params.get("required", [])
        args_doc = []
        for k, v in props.items():
            req   = " (required)" if k in required else ""
            typ   = v.get("type", "any")
            pdesc = v.get("description", "")
            args_doc.append(f"  - {k} ({typ}){req}: {pdesc}")
        tool_specs.append(
            f"### {name}\n{desc}\nArguments:\n"
            + ("\n".join(args_doc) if args_doc else "  (none)")
        )

    return (
        "\n\n---\n"
        "## Tool Calling\n\n"
        "You have access to tools. If you need to call one or more tools, "
        "output your full response text first, then append a tool call block "
        "as the VERY LAST thing in your message — no text after it.\n\n"
        "Format (strict — no markdown fences, no extra whitespace before the tag):\n"
        "<tool_calls>\n"
        '[{"name": "tool_name", "args": {"arg1": "val1"}}, ...]\n'
        "</tool_calls>\n\n"
        "Rules:\n"
        "- Only include the block if you are actually calling a tool.\n"
        "- All tool calls for a single turn go in one block as a JSON array.\n"
        "- args values must match the expected types exactly.\n"
        "- After you receive tool results they will appear as user messages; "
        "continue your response naturally.\n\n"
        "Available tools:\n\n"
        + "\n\n".join(tool_specs)
    )


def _inject_tool_system(messages: list, tools: list) -> list:
    tool_instructions = _build_tool_system(tools)
    msgs = list(messages)
    if msgs and msgs[0].get("role") == "system":
        msgs[0] = dict(msgs[0])
        msgs[0]["content"] = msgs[0]["content"] + tool_instructions
    else:
        msgs.insert(0, {"role": "system", "content": tool_instructions.strip()})
    return msgs


def _messages_to_input(messages: list) -> str:
    """Only type the last user message into the site."""
    for m in reversed(messages):
        if m.get("role") == "user":
            content = m.get("content") or ""
            if isinstance(content, list):
                content = " ".join(p.get("text", "") for p in content if p.get("type") == "text")
            return content.strip()
    return ""


def stream_chat(
    model_id: str,
    messages: list[dict],
    tools: list | None = None,
    max_tokens: int | None = None,
) -> Generator[dict, None, None]:
    from python.mediator.mediator_server import _get_executor, _load_manifest
    from python.mediator.script_session import session_manager
    from python.mediator.js_executor import JsError

    if tools:
        messages = _inject_tool_system(messages, tools)

    manifest = _load_manifest(SCRIPT_ID)
    if not manifest:
        yield {"type": "error", "text": "No manifest found for chatgpt script"}
        return

    executor = _get_executor(SCRIPT_ID)
    if not executor:
        yield {"type": "error", "text": "Failed to load chatgpt executor"}
        return

    user_input = _messages_to_input(messages)
    stream_selector = manifest.get("stream_selector", "")
    thinking_js     = manifest.get("thinking_snapshot_js", None)
    stable_ms       = manifest.get("timeouts", {}).get("stream_stable_ms",
                      manifest.get("timeouts", {}).get("stable_ms", 1500))

    try:
        if not session_manager.is_loaded(SCRIPT_ID):
            session_manager.get_or_create(
                script_id=SCRIPT_ID,
                url=manifest.get("url", ""),
                port=manifest.get("port", 0),
            )
            executor.load()
            session_manager.mark_loaded(SCRIPT_ID)

        executor.send(user_input)
        session_manager.reset_errors(SCRIPT_ID)

    except JsError as e:
        yield {"type": "error", "text": f"Script error: {e}"}
        return
    except Exception as e:
        yield {"type": "error", "text": f"Executor error: {e}"}
        return

    # ── Streaming: emit thinking + reply deltas ───────────────────────────────
    if stream_selector:
        reply_buffer = ""
        try:
            for item in executor.stream_read_with_thinking(
                stream_selector,
                thinking_js=thinking_js,
                stable_ms=stable_ms,
            ):
                chunk_type = item.get("type", "text")
                chunk_text = item.get("text", "")
                if chunk_type == "thinking":
                    yield {"type": "thinking", "text": chunk_text}
                else:
                    reply_buffer += chunk_text
                    yield {"type": "text", "text": chunk_text}
        except JsError as e:
            yield {"type": "error", "text": f"Stream error: {e}"}
            yield {"type": "done"}
            return

        reply = reply_buffer

    else:
        # Non-streaming fallback (no stream_selector in manifest)
        try:
            reply = executor.read() or ""
        except JsError as e:
            yield {"type": "error", "text": f"Script error: {e}"}
            return
        except Exception as e:
            yield {"type": "error", "text": f"Executor error: {e}"}
            return
        if reply:
            yield {"type": "text", "text": reply}

    # ── Tool call extraction ──────────────────────────────────────────────────
    tool_buffer = None
    if _TOOL_CALL_OPEN in reply:
        before, _, after = reply.partition(_TOOL_CALL_OPEN)
        tool_buffer = after.split(_TOOL_CALL_CLOSE)[0].strip()

    if tool_buffer is None:
        yield {"type": "done"}
        return

    try:
        tool_calls = json.loads(tool_buffer)
        if not isinstance(tool_calls, list):
            raise ValueError("expected JSON array")
    except Exception as e:
        yield {"type": "error", "text": f"Failed to parse tool_calls: {e}\nRaw: {tool_buffer}"}
        yield {"type": "done"}
        return

    for i, tc in enumerate(tool_calls):
        name = tc.get("name", "")
        args = tc.get("args", tc.get("arguments", {}))
        yield {
            "type":      "tool_delta",
            "index":     i,
            "id":        "",
            "name":      name,
            "arguments": json.dumps(args) if not isinstance(args, str) else args,
        }

    yield {"type": "done"}
